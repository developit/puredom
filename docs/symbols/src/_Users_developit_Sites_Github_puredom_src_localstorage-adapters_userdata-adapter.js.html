1 /**	@class Storage adapter that persists data into HTML5 LocalStorage.
  2  *	@name puredom.LocalStorage.adapters.UserData
  3  */
  4 puredom.LocalStorage.addAdapter('UserData', /** @lends puredom.LocalStorage.adapters.UserData */ {
  5 	
  6 	/** The default cookie ID to use for database storage */
  7 	defaultName : 'db',
  8 	
  9 	
 10 	/**	This adapter is a mediocre storage mechanism, so it gets a low rating. */
 11 	rating : 20,
 12 	
 13 	
 14 	/**	Test if this adapter will work in the current environment */
 15 	test : function(storage) {
 16 		// IE 6 and below crashes without an error description. Block it for now:
 17 		if ((/\bMSIE\s[1-6](\.[0-9]*)?/gim).test(navigator.userAgent+'')) {
 18 			return false;
 19 		}
 20 		return typeof document.body.addBehavior!=='undefined';
 21 	},
 22 	
 23 	
 24 	/**	Load the persisted DB */
 25 	load : function(storage, callback) {
 26 		var key = this._getKey(storage),
 27 			store = this._getStore(key),
 28 			json, data;
 29 		if (store) {
 30 			json = store.getAttribute('puredomlocalstorage');
 31 		}
 32 		if (json) {
 33 			data = puredom.json.parse(json);
 34 		}
 35 		if (callback) {
 36 			callback(data);
 37 		}
 38 		return data;
 39 	},
 40 	
 41 	
 42 	/**	Save the DB to UserData */
 43 	save : function(storage, data, callback) {
 44 		var key = this._getKey(storage),
 45 			store = this._getStore(key),
 46 			attr = 'puredomlocalstorage',
 47 			value,
 48 			saved = false;
 49 		if (store && ('save' in store)) {
 50 			if (data===undefined) {
 51 				if (store.removeAttribute) {
 52 					store.removeAttribute(attr);
 53 				}
 54 				else {
 55 					store.setAttribute(attr, '');
 56 				}
 57 			}
 58 			else {
 59 				store.setAttribute(attr, puredom.json.stringify(data));
 60 			}
 61 			store.save(key);
 62 			saved = true;
 63 		}
 64 		if (callback) {
 65 			callback(saved);
 66 		}
 67 		return saved;
 68 	},
 69 	
 70 	
 71 	/**	@private */
 72 	_getStore : function(key) {
 73 		var s;
 74 		if (!this.stores) {
 75 			this.stores = {};
 76 		}
 77 		s = this.stores[key];
 78 		if (!s) {
 79 			s = this.stores[key] = document.getElementById(key);
 80 			if (!s) {
 81 				s = this.stores[key] = document.createElement('span');
 82 				s.style.position = 'absolute';
 83 				s.style.top = '-100px';
 84 				s.style.left = '0';
 85 				s.style.behavior = "url('#default#userData')";
 86 				document.body.appendChild(s);
 87 			}
 88 		}
 89 		if (s.getAttribute('data-tdlsud-loaded')!=='true') {
 90 			s.setAttribute('data-tdlsud-loaded', 'true');
 91 			s.load(key);
 92 		}
 93 		return s;
 94 	},
 95 	
 96 	
 97 	/** Get the key for a storage object
 98 	 *	@private
 99 	 */
100 	_getKey : function(storage) {
101 		return 'ieud' + (storage.id || this.defaultName || '') + '';
102 	}
103 	
104 });
